from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.orm import Session
from sqlalchemy import func
from datetime import datetime

from app.deps import get_db, get_current_user
from app.models import Document, User, Organization
from app.schemas import DocCreate, DocUpdate

router = APIRouter(prefix="/documents", tags=["Documents"])


# ================= CREATE DOCUMENT =================
# ADMIN ONLY
@router.post("/")
def create_document(
    data: DocCreate,
    db: Session = Depends(get_db),
    user: User = Depends(get_current_user),
):
    # 🔒 Role check
    if user.role != "admin":
        raise HTTPException(status_code=403, detail="Only admin can create documents")

    # 🔒 Org validation (prevents FK crash)
    org = db.get(Organization, data.org_id)
    if not org:
        raise HTTPException(status_code=400, detail="Invalid org_id")

    doc = Document(
        # REQUIRED FIELDS (FROM PAYLOAD)
        title=data.title,
        content=data.content,
        doc_type=data.doc_type,
        doc_number=data.doc_number,
        file_url=data.file_url,
        hash=data.hash,
        issued_at=data.issued_at,
        target_role=data.target_role.lower(),
        org_id=data.org_id,

        # SYSTEM-CONTROLLED FIELDS (NOT FROM PAYLOAD)
        created_by=user.id,
        owner_id=user.id,
        created_at=datetime.utcnow(),
    )

    db.add(doc)
    db.commit()
    db.refresh(doc)

    # 🔢 RESPONSE SEQUENCE (STRICT)
    return {
        "id": doc.id,
        "owner_id": doc.owner_id,
        "doc_type": doc.doc_type,
        "doc_number": doc.doc_number,
        "file_url": doc.file_url,
        "hash": doc.hash,
        "issued_at": doc.issued_at,
        "created_at": doc.created_at,
    }


# ================= LIST DOCUMENTS =================
@router.get("/")
def list_documents(
    db: Session = Depends(get_db),
    user: User = Depends(get_current_user),
):
    query = (
        db.query(Document)
        .join(Organization, Organization.id == Document.org_id)
    )

    # 🔐 VISIBILITY RULES
    if user.role in ("admin", "auditor"):
        pass

    elif user.role in ("bank", "corp"):
        query = query.filter(
            Document.org_id == user.org_id,
            func.lower(Document.target_role).in_(["bank", "corp"]),
        )

    else:
        raise HTTPException(status_code=403, detail="Forbidden")

    docs = query.order_by(Document.created_at.desc()).all()

    # 🔢 RESPONSE SEQUENCE (STRICT)
    return [
        {
            "id": d.id,
            "owner_id": d.owner_id,
            "doc_type": d.doc_type,
            "doc_number": d.doc_number,
            "file_url": d.file_url,
            "hash": d.hash,
            "issued_at": d.issued_at,
            "created_at": d.created_at,
        }
        for d in docs
    ]


# ================= UPDATE DOCUMENT =================
# ADMIN ONLY
@router.put("/{doc_id}")
def update_document(
    doc_id: int,
    data: DocUpdate,
    db: Session = Depends(get_db),
    user: User = Depends(get_current_user),
):
    if user.role != "admin":
        raise HTTPException(status_code=403, detail="Only admin can update documents")

    doc = db.get(Document, doc_id)
    if not doc:
        raise HTTPException(status_code=404, detail="Document not found")

    for field, value in data.dict(exclude_unset=True).items():
        setattr(doc, field, value)

    db.commit()
    db.refresh(doc)

    return {"message": "Document updated successfully"}


# ================= DELETE DOCUMENT =================
# ADMIN ONLY
@router.delete("/{doc_id}")
def delete_document(
    doc_id: int,
    db: Session = Depends(get_db),
    user: User = Depends(get_current_user),
):
    if user.role != "admin":
        raise HTTPException(status_code=403, detail="Only admin can delete documents")

    doc = db.get(Document, doc_id)
    if not doc:
        raise HTTPException(status_code=404, detail="Document not found")

    db.delete(doc)
    db.commit()

    return {"message": "Document deleted successfully"}
